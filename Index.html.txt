// Hybrid: Hover or click keeps Services menu open, click toggles Investment Management submenu
(function() {
    const investmentToggle = document.getElementById('investment-management-toggle');
    const investmentSubmenu = document.getElementById('investment-management-submenu');
    let investmentOpen = false;

    // Investment submenu toggle (click only)
    investmentToggle.addEventListener('click', function(e) {
        e.preventDefault();
        investmentOpen = !investmentOpen;
        if (investmentOpen) {
            investmentSubmenu.classList.remove('opacity-0', 'pointer-events-none');
            investmentSubmenu.classList.add('opacity-100');
        } else {
            investmentSubmenu.classList.add('opacity-0', 'pointer-events-none');
            investmentSubmenu.classList.remove('opacity-100');
        }
    });

    // Close submenu when clicking outside (but not Services menu)
    document.addEventListener('mousedown', function(e) {
        if (!investmentSubmenu.contains(e.target) && !investmentToggle.contains(e.target)) {
            investmentSubmenu.classList.add('opacity-0', 'pointer-events-none');
            investmentSubmenu.classList.remove('opacity-100');
            investmentOpen = false;
        }
    });

    // Keyboard accessibility: close on Escape
    document.addEventListener('keydown', function(e) {
        if (e.key === 'Escape') {
            investmentSubmenu.classList.add('opacity-0', 'pointer-events-none');
            investmentSubmenu.classList.remove('opacity-100');
            investmentOpen = false;
        }
    });
})();
